pipeline {
	agent any

	stages {

		stage('Build'){
			steps {
				bat "mvnw clean install -DskipTests"
			}
		}

		stage('Test'){
			steps{
				bat "mvnw test"
			}
		}

		stage('Deploy') {
			steps {
			    sh "echo Deployment done"
			}
		}
		
		post {
		success {
			sh 'echo "Deployment success"'
			sh 'mvnw spring-boot:run'
		}
		failure {
			script {
				sh '''
				|echo "This job failed"
				'''.stripMargin().stripIndent()
			}
		}
}
	}
}
