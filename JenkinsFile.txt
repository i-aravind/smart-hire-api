pipeline {
	agent any

	stages {

		stage('Build'){
			steps {
				bat "mvnw clean install -DskipTests"
			}
		}

		stage('Test'){
			steps{
				bat "mvnw test"
			}
		}

		stage('Deploy') {
			steps {
			    bat "echo Deployment done"
			}
		}
	}
	
	post {
		success {
			bat 'echo "Deployment success"'
			bat 'echo Sending mail to jenkins.smarthire@gmail.com'
			emailext to: "jenkins.smarthire@gmail.com",
            subject: "Jenkins build success",
            body: "Jenins build success!!!"
		}
		failure {
			script {
				bat '''
				|echo "This job failed"
				'''.stripMargin().stripIndent()
			}
		}
	}
}
